git clone [repo-url] - clone remote repository

git add . - добавить все новые файлы в коммит
git commit -am 'тут название комита' - ты группируешь изменения и присваиваешь им заголовок
git push - отправить коммиты в удалённые репозиторий

git checkout -b 'BRANCH-NAME' - создание новой ветки
git checkout 'BRANCH-NAME' - переход на другую ветку
git branch - посмотреть все ветки
